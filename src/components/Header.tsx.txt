import { useAuth } from '@/contexts/AuthContext';
import { useRouter } from 'next/router';
import Link from 'next/link';
import styles from '@/styles/Header.module.css';

export function Header() {
  const { user, logout, isAuthenticated } = useAuth();
  const router = useRouter();

  // Não mostra o header na página de login
  if (router.pathname === '/login' || router.pathname === '/login-novo' || !isAuthenticated) {
    return null;
  }

  return (
    <header className={styles.header}>
      <div className={styles.container}>
        <div className={styles.logo}>
          <h2>Sistema de Gestão</h2>
        </div>
        
        <nav className={styles.nav}>
          <Link href="/" className={router.pathname === '/' ? styles.navLinkActive : styles.navLink}>
            Home
          </Link>
          <Link href="/matriculas" className={router.pathname === '/matriculas' ? styles.navLinkActive : styles.navLink}>
            Matrículas
          </Link>
          <Link href="/admin-usuarios" className={router.pathname === '/admin-usuarios' ? styles.navLinkActive : styles.navLink}>
            Usuários
          </Link>
        </nav>

        <div className={styles.userInfo}>
          <span className={styles.userEmail}>{user?.email}</span>
          <button onClick={logout} className={styles.logoutButton}>
            Sair
          </button>
        </div>
      </div>
    </header>
  );
}
